137int calcWidth(const std::string &str) {
138 int width = 0;
139 utf8proc_int32_t codepoint;
140 const utf8proc_uint8_t *utf8 = reinterpret_cast<const utf8proc_uint8_t *>(str.c_str());
141
142 while (*utf8) {
143 utf8proc_ssize_t result = utf8proc_iterate(utf8, -1, &codepoint);
144 if (result < 1) break;
145 utf8 += result;
146 width += utf8proc_charwidth(codepoint);
147 }
